設置方法

function routemap(url) {
  window.open(url, 'ROUTEMAP', 'width=640,height=510,resizable=yes');
  return false;
}
a[onclick] {
  font-size: smaller;
  margin-top: 0.5em;
  background-color: #fc9;
  text-decoration: none;
  border-width: 1px;
  border-style: solid;
  border-color: #fcc #f96 #f96 #fcc;
  border-radius: .4em;
}
<a href="omap_ex2.html?lat=35.913163&lon=138.721400&zoom=15&url=example/routemap.geojson" onclick="return routemap(this.href)">ルート地図</a>
